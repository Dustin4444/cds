### Basic usage

```tsx live noPreview
import { Button } from '@coinbase/cds-mobile/buttons';
import { PageFooter } from '@coinbase/cds-mobile/page';

function BasicPageFooter() {
  return (
    <PageFooter
      action={
        <Button variant="primary" onPress={() => console.log('Pressed')}>
          Submit
        </Button>
      }
    />
  );
}
```

### Multiple Actions

```tsx live noPreview
import { Button } from '@coinbase/cds-mobile/buttons';
import { HStack } from '@coinbase/cds-mobile/layout';
import { PageFooter } from '@coinbase/cds-mobile/page';

function MultipleActionsFooter() {
  return (
    <PageFooter
      action={
        <HStack gap={2} justifyContent="center">
          <Button variant="secondary" onPress={() => console.log('Cancel')}>
            Cancel
          </Button>
          <Button variant="primary" onPress={() => console.log('Submit')}>
            Submit
          </Button>
        </HStack>
      }
    />
  );
}
```

### With Background Color

```tsx live noPreview
import { Button } from '@coinbase/cds-mobile/buttons';
import { PageFooter } from '@coinbase/cds-mobile/page';

function StyledFooter() {
  return (
    <PageFooter
      action={
        <Button variant="primary" onPress={() => console.log('Pressed')}>
          Submit
        </Button>
      }
      backgroundColor="bgSecondary"
    />
  );
}
```
