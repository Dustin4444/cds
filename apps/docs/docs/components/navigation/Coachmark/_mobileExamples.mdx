### Basic usage

```tsx live noPreview
import { Button } from '@coinbase/cds-mobile/buttons';
import { Coachmark } from '@coinbase/cds-mobile/coachmark';

<Coachmark
  title="Welcome!"
  content="This is a Coachmark. Use it to guide users."
  action={<Button onPress={() => {}}>Got it</Button>}
  onClose={() => {}}
/>;
```

```tsx live noPreview
<Coachmark
  title="Feature Highlight"

```

### With Checkbox

```tsx live noPreview
import { Button } from '@coinbase/cds-mobile/buttons';
import { Coachmark } from '@coinbase/cds-mobile/coachmark';
import { Checkbox } from '@coinbase/cds-mobile/controls';

<Coachmark
  title="Don't show again?"
  content="You can add a checkbox to let users opt out."
  checkbox={
    <Checkbox value={false} onChange={() => {}}>
      Don't show this again
    </Checkbox>
  }
  action={<Button onPress={() => {}}>Close</Button>}
  onClose={() => {}}
/>;
```
