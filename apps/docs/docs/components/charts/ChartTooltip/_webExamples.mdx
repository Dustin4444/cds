ChartTooltip renders values for the currently highlighted data index in a floating panel. It is typically used with [LineChart](/components/charts/LineChart) or [BarChart](/components/charts/BarChart), and works best when chart highlighting is enabled.

## Basics

You can add `ChartTooltip` as a child of with highlighting enabled so the tooltip has an active index to render.

```jsx live
<LineChart
  enableScrubbing
  legend
  showArea
  showXAxis
  showYAxis
  height={{ base: 200, tablet: 225, desktop: 250 }}
  legendPosition="right"
  series={[
    {
      id: 'pageViews',
      data: [2400, 1398, 9800, 3908, 4800, 3800, 4300],
      color: 'rgb(var(--green40))',
      label: 'Page Views',
    },
    {
      id: 'uniqueVisitors',
      data: [4000, 3000, 2000, 2780, 1890, 2390, 3490],
      color: 'rgb(var(--purple40))',
      label: 'Unique Visitors',
      areaType: 'dotted',
    },
  ]}
  xAxis={{
    data: ['Page A', 'Page B', 'Page C', 'Page D', 'Page E', 'Page F', 'Page G'],
  }}
  yAxis={{
    position: 'left',
    showGrid: true,
    tickLabelFormatter: (value) =>
      new Intl.NumberFormat('en-US', { maximumFractionDigits: 0 }).format(value),
  }}
>
  <Scrubber hideBeaconLabels />
  <ChartTooltip
    valueFormatter={(value) =>
      new Intl.NumberFormat('en-US', { maximumFractionDigits: 0 }).format(value)
    }
  />
</LineChart>
```

## Filtering Series

Use `seriesIds` to control which series are shown in the tooltip.

```jsx live
function TooltipSeriesFilter() {
  const days = useMemo(() => ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'], []);
  const prices = useMemo(() => [102, 104, 101, 108, 110, 111, 109], []);
  const volume = useMemo(() => [32, 44, 30, 52, 47, 60, 55], []);

  return (
    <LineChart
      enableHighlighting
      showArea
      showXAxis
      showYAxis
      height={260}
      series={[
        { id: 'price', label: 'Price', data: prices, color: 'var(--color-accentBoldBlue)' },
        { id: 'volume', label: 'Volume', data: volume, color: 'var(--color-accentBoldGreen)' },
      ]}
      xAxis={{ data: days }}
    >
      <Scrubber hideBeaconLabels />
      <ChartTooltip
        label={(index) => `Day ${index + 1}`}
        seriesIds={['price']}
        valueFormatter={(value) => {
          if (value === null || value === undefined) return '-';
          if (Array.isArray(value)) return `${value[0]}-${value[1]}`;
          return new Intl.NumberFormat('en-US').format(value);
        }}
      />
    </LineChart>
  );
}
```

## Styles

### Sorting

You can adjust `sort` to change the order of entries in the tooltip. Ordering uses each series' raw `data` value at the highlighted index (the same values passed to `valueFormatter`), not stacked plot coordinates from `getSeriesData`.

```jsx live
function TooltipSortOrder() {
  const sortTabs = [
    { id: 'none', label: 'none' },
    { id: 'ascending', label: 'Ascending' },
    { id: 'descending', label: 'Descending' },
  ];
  const [sortTab, setSortTab] = useState(sortTabs[0]);
  const days = useMemo(() => ['Mon', 'Tue', 'Wed', 'Thu', 'Fri'], []);
  const seriesA = useMemo(() => [30, null, 10, 25, 5], []);
  const seriesB = useMemo(() => [10, 20, 5, 15, 40], []);
  const seriesC = useMemo(() => [20, 40, null, 10, 12], []);

  return (
    <VStack gap={2}>
      <HStack alignItems="center" gap={2} justifyContent="flex-end">
        <Text as="h3" font="headline">
          Sorting
        </Text>
        <SegmentedTabs activeTab={sortTab} onChange={setSortTab} tabs={sortTabs} />
      </HStack>
      <LineChart
        enableHighlighting
        height={260}
        series={[
          { id: 'a', label: 'Series A', data: seriesA, color: 'var(--color-accentBoldBlue)' },
          { id: 'b', label: 'Series B', data: seriesB, color: 'var(--color-accentBoldGreen)' },
          { id: 'c', label: 'Series C', data: seriesC, color: 'var(--color-accentBoldOrange)' },
        ]}
        showArea
        showXAxis
        showYAxis
        xAxis={{ data: days }}
      >
        <Scrubber hideBeaconLabels />
        <ChartTooltip
          sort={sortTab.id}
          valueFormatter={(value) => {
            if (value === null || value === undefined) return '—';
            if (Array.isArray(value)) return `${value[0]}–${value[1]}`;
            return String(value);
          }}
        />
      </LineChart>
    </VStack>
  );
}
```

## Custom Value Formatting

Since `valueFormatter` receives `number | [number, number] | null`, you can format both scalar values and tuple ranges however you want.

### Range Values

You can pass in `[min, max]` tuples as data points and format them as currency ranges in the tooltip.

```jsx live
function TooltipRangeValues() {
  const candles = btcCandles.slice(0, 120).reverse();
  const data = candles.map((candle) => [parseFloat(candle.low), parseFloat(candle.high)]);
  const dates = candles.map((candle) =>
    new Date(Number(candle.start) * 1000).toLocaleDateString('en-US', {
      month: 'short',
      day: 'numeric',
    }),
  );

  const min = Math.min(...data.map(([low]) => low));
  const max = Math.max(...data.map(([, high]) => high));

  const compactCurrency = useCallback(
    (value) =>
      new Intl.NumberFormat('en-US', {
        style: 'currency',
        currency: 'USD',
        notation: 'compact',
        maximumFractionDigits: 0,
      }).format(value),
    [],
  );

  return (
    <BarChart
      enableHighlighting
      fadeOnHighlight
      height={250}
      series={[{ id: 'prices', label: 'Daily price range', data, color: assets.btc.color }]}
      xAxis={{ data: dates }}
      showYAxis
      yAxis={{ domain: { min, max }, showGrid: true, tickLabelFormatter: compactCurrency }}
    >
      <ChartTooltip
        valueFormatter={(value) => {
          if (!Array.isArray(value)) return value ?? '-';
          return `${compactCurrency(value[0])} - ${compactCurrency(value[1])}`;
        }}
      />
    </BarChart>
  );
}
```

### Negative Values

For positive/negative data, you can render signs and units directly in the tooltip.

```jsx live
function TooltipNegativeValues() {
  const categories = Array.from({ length: 12 }, (_, i) => `3/${i + 1}`);
  const gains = [5, 0, 6, 18, 0, 5, 12, 0, 12, 22, 28, 18];
  const losses = [-4, 0, -8, -12, -6, 0, 0, 0, -18, 0, -12, 0];

  return (
    <BarChart
      enableHighlighting
      fadeOnHighlight
      height={250}
      series={[
        { id: 'gains', data: gains, color: 'var(--color-fgPositive)' },
        { id: 'losses', data: losses, color: 'var(--color-fgNegative)' },
      ]}
      showXAxis
      showYAxis
      xAxis={{ data: categories }}
      yAxis={{ showGrid: true, tickLabelFormatter: (value) => `$${value}M` }}
    >
      <ChartTooltip
        valueFormatter={(value) => {
          if (value === null || value === undefined) return '$0M';
          if (Array.isArray(value)) return `${value[0]}-${value[1]}`;
          const sign = value > 0 ? '+' : '';
          return `${sign}$${value}M`;
        }}
      />
    </BarChart>
  );
}
```

## Accessibility

ChartTooltip is visual-only, so pair it with chart-level `accessibilityLabel` and scrubber labels for screen readers.

```jsx live
function AccessibleTooltip() {
  const days = useMemo(() => ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'], []);
  const prices = useMemo(() => [102, 104, 101, 108, 110, 111, 109], []);

  const getChartAccessibilityLabel = useCallback(
    (item) => {
      if (item.dataIndex === null) return 'Interacting with chart';
      return `${days[item.dataIndex]} price is ${prices[item.dataIndex]} dollars`;
    },
    [days, prices],
  );

  return (
    <LineChart
      enableHighlighting
      showArea
      showXAxis
      showYAxis
      accessibilityLabel={getChartAccessibilityLabel}
      height={260}
      series={[{ id: 'price', label: 'Price', data: prices, color: 'var(--color-accentBoldBlue)' }]}
      xAxis={{ data: days }}
    >
      <Scrubber hideBeaconLabels />
      <ChartTooltip
        valueFormatter={(value) => {
          if (value === null || value === undefined) return '-';
          if (Array.isArray(value)) return `${value[0]}-${value[1]}`;
          return `$${value}`;
        }}
      />
    </LineChart>
  );
}
```
